-------------sw-ins-core-------------
ena
conf t
host SW-INS-Core
vtp mode server
vtp domain bkacad
vtp pass abc
vtp ver 2

vlan 10
name tech
exit
vlan 20
name INS
exit
vlan 30
name staff
exit
vlan 40
name student
exit

int vlan 10
ip add 192.168.10.1 255.255.255.0
ip help 10.0.2.2
ex
int vlan 20
ip add 192.168.20.1 255.255.255.0
ip help 10.0.2.2
ex

int f0/1
no sw
ip add 10.0.1.2 255.255.255.0
no shut
ex

int f0/2
sw trunk enc dot1q
sw mo trunk
ex

int f0/4
sw trunk enc dot1q
sw mo trunk
ex


ip routing
ip route 0.0.0.0 0.0.0.0 10.0.1.1
do wr
